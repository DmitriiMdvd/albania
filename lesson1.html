      <div id="score">Очки: 0</div>

    </div>

    <button id="nextBtn" style="display:none;">Следующий</button>

  </div>


  <script>

    const questions = [

      {

        letter: "Ç",

        correct: "ч",

        choices: ["к", "ч", "дж"]

      },

      {

        letter: "Dh",

        correct: "ð (как англ. this)",

        choices: ["θ", "ð (как англ. this)", "ш"]

      },

      {

        letter: "Ll",

        correct: "долгое л",

        choices: ["обычное л", "долгое л", "р"]

      },

      {

        letter: "Y",

        correct: "ü (как немецкое)",

        choices: ["у", "и", "ü (как немецкое)"]

      },

    ];


    let current = 0;

    let score = 0;


    const letterEl = document.getElementById("letter");

    const choicesEl = document.getElementById("choices");

    const scoreEl = document.getElementById("score");

    const nextBtn = document.getElementById("nextBtn");


    function showQuestion() {

      const q = questions[current];

      letterEl.textContent = q.letter;

      choicesEl.innerHTML = "";

      nextBtn.style.display = "none";


      q.choices.forEach(choice => {

        const btn = document.createElement("div");

        btn.className = "choice";

        btn.textContent = choice;

        btn.onclick = () => checkAnswer(btn, choice === q.correct);

        choicesEl.appendChild(btn);

      });

    }


    function checkAnswer(btn, correct) {

      const all = document.querySelectorAll(".choice");

      all.forEach(c => c.onclick = null);

      btn.classList.add(correct ? "correct" : "wrong");

      if (correct) score++;

      scoreEl.textContent = "Очки: " + score;

      nextBtn.style.display = "inline-block";

    }


    nextBtn.onclick = () => {

      current++;

      if (current < questions.length) {

        showQuestion();

      } else {

        letterEl.textContent = "🎉";

        choicesEl.innerHTML = `<p>Урок завершён! Вы набрали ${score} из ${questions.length} очков.</p>`;

        nextBtn.style.display = "none";

      }

    };


    showQuestion();

  </script>
